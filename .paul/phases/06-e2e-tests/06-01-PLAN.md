  <action>
    Créer `e2e/auth.spec.ts` :

    ```typescript
    import { test, expect } from '@playwright/test'

    const EMAIL = 'e2e-test@test.com'
    const PASSWORD = 'Test1234!'

    test.describe('Auth', () => {
      test('login → dashboard → logout', async ({ page }) => {
        await page.goto('/auth')
        // Essayer login d'abord
        await page.getByLabel(/email/i).fill(EMAIL)
        await page.getByLabel(/mot de passe|password/i).fill(PASSWORD)
        await page.getByRole('button', { name: /connexion|se connecter|login/i }).click()

        // Si erreur "utilisateur non trouvé" → register
        const errorVisible = await page.getByRole('alert').isVisible().catch(() => false)
        if (errorVisible) {
          // Basculer sur l'onglet Register
          await page.getByRole('button', { name: /créer|register|inscription/i }).click()
          await page.getByLabel(/email/i).fill(EMAIL)
          await page.getByLabel(/mot de passe|password/i).fill(PASSWORD)
          await page.getByRole('button', { name: /créer|register|s'inscrire/i }).click()
        }

        // Doit être sur /dashboard
        await expect(page).toHaveURL(/dashboard/)
        await expect(page.getByRole('heading', { name: /tableau de bord|dashboard/i })).toBeVisible()

        // Logout
        await page.getByRole('button', { name: /déconnexion|logout/i }).click()
        await expect(page).toHaveURL(/auth/)
      })
    })
    ```

    Utiliser des sélecteurs accessibles (getByLabel, getByRole) — pas de sélecteurs CSS fragiles.
    Si les libellés exacts diffèrent du code, ajuster les regex pour matcher ce que AnalysisPage/AuthPage affichent réellement (lire les fichiers sources avant).

    **e2e/analysis.spec.ts** :
    ```typescript
    import { test, expect } from '@playwright/test'
    import { loginAs } from './fixtures/auth'

    test.describe('Analysis', () => {
      test.beforeEach(async ({ page }) => {
        await loginAs(page)
        await page.goto('/analysis')
      })

      test('page charge avec sélecteur actif', async ({ page }) => {
        await expect(page.getByRole('heading', { name: /analyse/i })).toBeVisible()
        await expect(page.locator('select')).toBeVisible()
      })

      test('Prédire AAPL → affiche une valeur', async ({ page }) => {
        // Sélectionner AAPL
        await page.selectOption('select', { label: /AAPL/i })
        await page.getByRole('button', { name: /prédire/i }).click()

        // Attendre le résultat (loading puis valeur ou erreur gérée)
        await page.waitForFunction(() => {
          const btns = document.querySelectorAll('button')
          return !Array.from(btns).some((b) => b.getAttribute('aria-busy') === 'true')
        }, { timeout: 20000 })

        // Soit une valeur numérique, soit une erreur propre (pas de crash page blanche)
        const hasValue = await page.getByText(/USD/i).isVisible().catch(() => false)
        const hasError = await page.getByRole('alert').isVisible().catch(() => false)
        expect(hasValue || hasError).toBe(true)
      })
    })
    ```
