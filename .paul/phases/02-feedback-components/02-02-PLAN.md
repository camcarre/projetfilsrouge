  <files>src/contexts/ToastContext.tsx</files>
  <action>
    Créer `src/contexts/ToastContext.tsx` :

    ```tsx
    import { createContext } from 'preact'
    import { useContext, useState, useCallback } from 'preact/hooks'
    import type { ComponentChildren } from 'preact'

    type ToastVariant = 'default' | 'success' | 'error'

    interface Toast {
      id: number
      message: string
      variant: ToastVariant
    }

    interface ToastContextValue {
      toasts: Toast[]
      toast: (message: string, variant?: ToastVariant) => void
      dismiss: (id: number) => void
    }

    const ToastContext = createContext<ToastContextValue | null>(null)

    let nextId = 0

    export function ToastProvider({ children }: { children: ComponentChildren }) {
      const [toasts, setToasts] = useState<Toast[]>([])

      const dismiss = useCallback((id: number) => {
        setToasts((prev) => prev.filter((t) => t.id !== id))
      }, [])

      const toast = useCallback((message: string, variant: ToastVariant = 'default') => {
        const id = nextId++
        setToasts((prev) => [...prev, { id, message, variant }])
        setTimeout(() => dismiss(id), 2000)
      }, [dismiss])

      return (
        <ToastContext.Provider value={{ toasts, toast, dismiss }}>
          {children}
        </ToastContext.Provider>
      )
    }

    export function useToast() {
      const ctx = useContext(ToastContext)
      if (!ctx) throw new Error('useToast must be used within ToastProvider')
      return ctx.toast
    }
    ```

    Named exports uniquement. Pas de useEffect pour le timer (setTimeout direct dans le callback).
    Ne pas importer depuis 'react' — utiliser 'preact' et 'preact/hooks'.
  </action>
  <verify>Importer `{ useToast, ToastProvider }` dans un fichier TSX → tsc sans erreur.</verify>

  <files>src/components/ui/Toast.tsx</files>
  <action>
    Créer `src/components/ui/Toast.tsx` :

    ```tsx
    import { useContext } from 'preact/hooks'
    import { createContext } from 'preact'
    import { ToastContext } from '@/contexts/ToastContext'

    const VARIANT_CLASSES = {
      default: 'bg-neutral-800 dark:bg-neutral-700 text-neutral-100',
      success: 'bg-emerald-600 dark:bg-emerald-700 text-white',
      error:   'bg-red-600 dark:bg-red-700 text-white',
    }

    export function ToastContainer() {
      const ctx = useContext(ToastContext)
      if (!ctx || ctx.toasts.length === 0) return null

      return (
        <div
          role="region"
          aria-label="Notifications"
          aria-live="polite"
          className="fixed bottom-20 right-4 z-[200] flex flex-col gap-2 lg:bottom-6"
        >
          {ctx.toasts.map((t) => (
            <div
              key={t.id}
              className={`rounded-lg px-4 py-2.5 text-sm font-medium shadow-lg max-w-xs ${VARIANT_CLASSES[t.variant]}`}
            >
              {t.message}
            </div>
          ))}
        </div>
      )
    }
    ```

    Positionné `bottom-20` sur mobile (au-dessus de la bottom nav) et `bottom-6` sur lg+.
    `ToastContext` doit être exporté depuis ToastContext.tsx pour que ce fichier puisse l'importer.
    Ajouter `export { ToastContext }` dans ToastContext.tsx si ce n'est pas déjà fait.
  </action>
